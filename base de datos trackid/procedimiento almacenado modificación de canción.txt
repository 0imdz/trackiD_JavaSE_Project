DELIMITER |
 
    spLanzamientoModified(
	in PUPC INT(11),
	in PTITULO VARCHAR(200),
	in PAUTORIA VARCHAR(200),
	in PGENERO VARCHAR(20),
	in PFECHA_LANZAMIENTO DATE,
    	in PSELLO VARCHAR(100),
    	in PC_EXPLICITO VARCHAR(20),
    	in PDURACION INT(11),
   	in PAUDIO VARCHAR(200),
    	in PIMAGEN VARCHAR(200)
)
BEGIN
	UPDATE Cancion SET 
    	UPC=PUPC,
	TITULO=PTITULO,
	AUTORIA=PAUTORIA,
	GENERO=PGENERO,
	FECHA_LANZAMIENTO=PFECHA_LANZAMIENTO,
    	SELLO=PSELLO,
	C_EXPLICITO=PC_EXPLICITO,
    	DURACION=PDURACION,
    	AUDIO=PAUDIO,
    	IMAGEN=PIMAGEN
    		where 
    		upc=pupc;
END|
DELIMITER;
